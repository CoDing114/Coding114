        function fetchPage(name){
            fetch(name).then(function(response){ 
                response.text().then(function(text){
                document.getElementById('contents').innerHTML=text;
                })
            });
        } 
            if(location.hash){
            fetchPage(location.hash.substr(2));
        } else{
            fetchPage('data/welcome');
        } 
            fetch('data/list_v2').then(function(response){
            response.text().then(function(text){
            var items=text.split(','); 
            var i=0; 
            var tags=''; 
            while(i<items. length){
                var item=items[i]; 
                item=item.trim(); 
                var tag='<li><a href="#data/'+item+'"onclick="fetchPage(\'data/'+item+'\')">'+item+'</a></li>'; 
                tags=tags+tag; 
                i=i+1; 
            }
            document.getElementById('nav').innerHTML=tags;
            })
        });
